{

    input(_AAA);
    input(_BBB);
    input(_CCC);

    if (_AAA gt _BBB && _AAA gt _CCC) { _AAA -> _MAX; } else ;
    if (_BBB gt _AAA && _BBB gt _CCC) { _BBB -> _MAX; } else ;
    if (_CCC gt _AAA && _CCC gt _BBB) { _CCC -> _MAX; } else ;
    output(_MAX);

    ?? усі числа однакові
    if ((_AAA eg _BBB) && (_BBB eg _CCC)) { output(1); } else { output(0); }

    ?? хоча б одне від’ємне
    if ((_AAA lt 0) || (_BBB lt 0) || (_CCC lt 0)) { output(-1); } else { output(0); }

    ?? перевірка A > B + C
    if (_AAA gt (_BBB + _CCC)) { output(10); } else { output(0); }
