
program _PRG;
var int _AAA , _BBB , _CCC , _MAX ;
begin
{
  get ( _AAA )
  get ( _BBB )
  get ( _CCC )

  if ( _AAA gt _BBB && _AAA gt _CCC ) {
    _AAA -> _MAX ;
  } else if ( _BBB gt _AAA && _BBB gt _CCC ) {
    _BBB -> _MAX ;
  } else {
    _CCC -> _MAX ;
  }
  put ( _MAX )

  if ( _AAA eg _BBB && _AAA eg _CCC ) {
    put ( 1 ) ;
  } else {
    put ( 0 ) ;
  }

  if ( _AAA lt 0 || _BBB lt 0 || _CCC lt 0 ) {
    put ( -1 ) ;
  } else {
    put ( 0 ) ;
  }

  if !! ( _AAA lt ( _BBB + _CCC ) ) {
    put ( 10 ) ;
  } else {
    put ( 0 ) ;
  }
}

end
