name _PROGRAM2 ;
body data integer16 _AAA , _BBB , _CCC , _MAX ;

  get ( _AAA )
  get ( _BBB )
  get ( _CCC )

  if _AAA > _BBB and _AAA > _CCC {
    _AAA -> _MAX
  } else if _BBB > _AAA and _BBB > _CCC {
    _BBB -> _MAX
  } else {
    _CCC -> _MAX
  }
  put ( _MAX )

  if ( _AAA == _BBB ) and ( _AAA == _CCC ) {
    put ( 1 )
  } else {
    put ( 0 )
  }

  if ( _AAA < 0 ) or ( _BBB < 0 ) or ( _CCC < 0 ) {
    put ( -1 )
  } else {
    put ( 0 )
  }

  if not ( _AAA < ( _BBB + _CCC ) ) {
    put ( 10 )
  } else {
    put ( 0 )
  }

end
