program _PRG;
var int _AAA , _BBB , _CYA , _CYB , _XVA ;
begin

  get ( _AAA )
  get ( _BBB )

  _AAA -> _CYA
_LCBEGIN:
  if _CYA > _BBB {
    goto _LCEND
  }
  put ( _CYA * _CYA )
  _CYA + 1 -> _CYCLEA
  goto _LCBEGIN
_LCEND:

  0 -> _XVA
  1 -> _CYA
_LABEGIN:
  if _CYA > _AAA {
    goto _LAEND
  }
  1 -> _CYB
_LBBEGIN:
  if _CYB > _BBB {
    goto _LBEND
  }
  _XVA + 1 -> _XVA
  _CYB + 1 -> _CYB
  goto _LBBEGIN
_LBEND:
  _CYA + 1 -> _CYA
  goto _LABEGIN
_LAEND:

  put ( _XVA )

end
