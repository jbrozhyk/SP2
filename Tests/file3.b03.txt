name _PROGRAM3 ;
body data integer16 _AAA , _BBB , _CYCLEA , _CYCLEB , _XVALUE ;

  get ( _AAA )
  get ( _BBB )

  _AAA -> _CYCLEA
_LCBEGIN:
  if _CYCLEA > _BBB {
    goto _LCEND
  }
  put ( _CYCLEA * _CYCLEA )
  _CYCLEA + 1 -> _CYCLEA
  goto _LCBEGIN
_LCEND:

  0 -> _XVALUE
  1 -> _CYCLEA
_LABEGIN:
  if _CYCLEA > _AAA {
    goto _LAEND
  }
  1 -> _CYCLEB
_LBBEGIN:
  if _CYCLEB > _BBB {
    goto _LBEND
  }
  _XVALUE + 1 -> _XVALUE
  _CYCLEB + 1 -> _CYCLEB
  goto _LBBEGIN
_LBEND:
  _CYCLEA + 1 -> _CYCLEA
  goto _LABEGIN
_LAEND:

  put ( _XVALUE )

end
